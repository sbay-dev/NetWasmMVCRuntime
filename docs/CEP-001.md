| الحقل | القيمة |

## 1. الملخص التنفيذي

يُقدِّم هذا المقترح ثلاثة أنظمة متكاملة لتطوير **NetWasmMVCRuntime** إلى منصة حافة متكاملة:

1. **`CephaFS`** — نظام ملفات افتراضي يعمل في Node.js و Cloudflare Workers بواجهة موحدة
2. **طوبولوجيا التنفيذ** — إعلان صريح لمكان تنفيذ كل Controller بسطر واحد
3. **تعزيزات أمنية** — مُصمَّمة خصيصاً لبيئة WASM على الحافة

الهدف المحوري: **الحفاظ التام على نهج ASP.NET Core MVC** مع تمكين النشر على Cloudflare Workers بدون تكلفة — المسار الأسرع لتطبيقات MVC الرصينة.

## 3. `CephaFS` — نظام الملفات الافتراضي

### 3.1 المبدأ المعماري

```
المتصفح استوحى الحل منذ البداية:
────────────────────────────────
Browser: SQLite → OPFS (binary blob في IndexedDB)
         السطر: await JsInterop.PersistDbToOPFS(base64)

CephaFS يُعمِّم هذا المبدأ:
────────────────────────────
Node.js:    SQLite → disk مباشرة
Cloudflare: SQLite → KV Store (binary blob, base64)
            نفس الـ CephaInterop.StorageGet/Set الموجودة
```

### 3.2 هيكل `CephaFS`

```
WasmMvcRuntime.Cepha/
└── FS/
    ├── ICephaFileProvider.cs       ← العقد الموحد
    ├── CephaFileEntry.cs           ← تمثيل الملف (binary + MIME + version)
    ├── CephaFileVersion.cs         ← سجل الإصدارات المشفَّر
    ├── NodeFileProvider.cs         ← Node.js: disk مباشر
    ├── CloudflareKVFileProvider.cs ← Cloudflare: KV/R2
    ├── MemoryFileProvider.cs       ← Testing/ephemeral
    ├── CephaSecretProvider.cs      ← إدارة الأسرار
    └── CephaDbFileManager.cs       ← SQLite كـ stream
```

### 3.4 `CephaFileEntry` — الملف كـ Stream

```csharp
public sealed class CephaFileEntry
{
    // ── الهوية ─────────────────────────────────────────────
    public string   Path      { get; init; } = "";
    public string   VersionId { get; init; } =
        Guid.NewGuid().ToString("N")[..8];

    // ── المحتوى ────────────────────────────────────────────
    public byte[]   Data      { get; init; } = Array.Empty<byte>();
    public string   MimeType  { get; init; } = "application/octet-stream";
    public long     SizeBytes => Data.Length;

    // ── البيانات الوصفية ────────────────────────────────────
    public DateTime CreatedAt  { get; init; } = DateTime.UtcNow;
    public DateTime ModifiedAt { get; init; } = DateTime.UtcNow;
    public string?  CreatedBy  { get; init; }
    public string?  Hash       { get; init; } // SHA-256 hex

    // ── تصنيف المحتوى ─────────────────────────────────────
    public static string DetectMime(string path) => path switch
    {
        _ when path.EndsWith(".db")     => "application/x-sqlite3",
        _ when path.EndsWith(".db-wal") => "application/x-sqlite3-wal",
        _ when path.EndsWith(".json")   => "application/json",
        _ when path.EndsWith(".env")    => "text/plain",
        _ when path.EndsWith(".pem")    => "application/x-pem-file",
        _ when path.EndsWith(".pfx")    => "application/x-pkcs12",
        _ when path.EndsWith(".key")    => "application/x-pem-file",
        _ when path.EndsWith(".wasm")   => "application/wasm",
        _ when path.EndsWith(".html")   => "text/html",
        _ when path.EndsWith(".css")    => "text/css",
        _ when path.EndsWith(".js")     => "application/javascript",
        _                               => "application/octet-stream"
    };

    // ── التحويل ────────────────────────────────────────────
    public static CephaFileEntry FromBytes(string path, byte[] data)
    {
        using var sha = System.Security.Cryptography.SHA256.Create();
        var hash = Convert.ToHexString(sha.ComputeHash(data)).ToLower();
        return new CephaFileEntry
        {
            Path     = path,
            Data     = data,
            MimeType = DetectMime(path),
            Hash     = hash
        };
    }

    public static CephaFileEntry FromText(string path, string text) =>
        FromBytes(path, System.Text.Encoding.UTF8.GetBytes(text));

    public string AsText() =>
        System.Text.Encoding.UTF8.GetString(Data);

    // ── Serialization للـ KV Store ─────────────────────────
    // تنسيق: header_base64 | data_base64
    public string ToKvValue()
    {
        var header = System.Text.Json.JsonSerializer.Serialize(new
        {
            mime    = MimeType,
            size    = SizeBytes,
            created = CreatedAt,
            modified= ModifiedAt,
            version = VersionId,
            hash    = Hash
        });
        return $"{Convert.ToBase64String(
                   System.Text.Encoding.UTF8.GetBytes(header)
               )}|{Convert.ToBase64String(Data)}";
    }

    public static CephaFileEntry FromKvValue(string path, string kv)
    {
        var parts = kv.Split('|', 2);
        var data  = parts.Length > 1
            ? Convert.FromBase64String(parts[1])
            : Array.Empty<byte>();
        return FromBytes(path, data);
    }
}
```

### 3.6 `CloudflareKVFileProvider` — التطبيق الكامل

```csharp
public class CloudflareKVFileProvider : ICephaFileProvider
{
    private const string FilePrefix    = "cepha:file:";
    private const string MetaPrefix    = "cepha:meta:";
    private const int    MaxVersions   = 10; // احتفظ بآخر 10 إصدارات

    public CephaFileEnvironment Environment => CephaFileEnvironment.CloudflareKV;

    // ── مفاتيح KV ─────────────────────────────────────────
    private static string FileKey(string path) =>
        $"{FilePrefix}{path.TrimStart('/').Replace('/', ':')}";

    private static string MetaKey(string path) =>
        $"{MetaPrefix}{path.TrimStart('/').Replace('/', ':')}";

    // ── القراءة ────────────────────────────────────────────
    public Task<CephaFileEntry?> GetAsync(string path)
    {
        var raw = CephaInterop.StorageGet(FileKey(path));
        if (string.IsNullOrEmpty(raw))
            return Task.FromResult<CephaFileEntry?>(null);

        var entry = CephaFileEntry.FromKvValue(path, raw);
        return Task.FromResult<CephaFileEntry?>(entry);
    }

    // ── الكتابة مع إصدار تلقائي ────────────────────────────
    public async Task PutAsync(CephaFileEntry entry)
    {
        // 1. احفظ نسخة الإصدار السابق
        var existing = await GetAsync(entry.Path);
        if (existing != null)
            await ArchiveVersionAsync(existing);

        // 2. احفظ الملف الجديد
        CephaInterop.StorageSet(FileKey(entry.Path), entry.ToKvValue());

        // 3. حدِّث الفهرس للتصفح
        await UpdateIndexAsync(entry);
    }

    // ── أرشفة إصدار ────────────────────────────────────────
    private async Task ArchiveVersionAsync(CephaFileEntry old)
    {
        var versionKey = CephaFileVersion.KvKey(
            old.Path, old.VersionId);
        CephaInterop.StorageSet(versionKey, old.ToKvValue());

        // تحديث فهرس الإصدارات
        var indexKey = CephaFileVersion.IndexKey(old.Path);
        var indexRaw = CephaInterop.StorageGet(indexKey) ?? "[]";
        var versions = System.Text.Json.JsonSerializer
            .Deserialize<List<string>>(indexRaw) ?? new();

        versions.Insert(0, old.VersionId);

        // احذف الإصدارات القديمة تجاوزت الحد
        while (versions.Count > MaxVersions)
        {
            var obsolete = versions[^1];
            versions.RemoveAt(versions.Count - 1);
            CephaInterop.StorageRemove(
                CephaFileVersion.KvKey(old.Path, obsolete));
        }

        CephaInterop.StorageSet(indexKey,
            System.Text.Json.JsonSerializer.Serialize(versions));

        await Task.CompletedTask;
    }

    // ── استرجاع إصدار سابق ─────────────────────────────────
    public async Task<CephaFileEntry?> RestoreVersionAsync(
        string path, string versionId)
    {
        var versionKey = CephaFileVersion.KvKey(path, versionId);
        var raw = CephaInterop.StorageGet(versionKey);
        if (string.IsNullOrEmpty(raw))
            return null;

        var entry = CephaFileEntry.FromKvValue(path, raw);
        // استعادة = كتابة الإصدار القديم كإصدار جديد
        await PutAsync(entry);
        return entry;
    }

    // ── قائمة الإصدارات ────────────────────────────────────
    public Task<IReadOnlyList<CephaFileVersion>> GetVersionsAsync(
        string path, int limit = 10)
    {
        var indexKey = CephaFileVersion.IndexKey(path);
        var indexRaw = CephaInterop.StorageGet(indexKey) ?? "[]";
        var versionIds = System.Text.Json.JsonSerializer
            .Deserialize<List<string>>(indexRaw) ?? new();

        var result = versionIds
            .Take(limit)
            .Select(vid => new CephaFileVersion
            {
                VersionId = vid,
                Path      = path,
                CreatedAt = DateTime.UtcNow // مُبسَّط
            })
            .ToList();

        return Task.FromResult<IReadOnlyList<CephaFileVersion>>(result);
    }

    // ── تصفح الملفات ───────────────────────────────────────
    // الفهرس: cepha:index → JSON array of paths
    private const string GlobalIndex = "cepha:index";

    private async Task UpdateIndexAsync(CephaFileEntry entry)
    {
        var raw   = CephaInterop.StorageGet(GlobalIndex) ?? "[]";
        var paths = System.Text.Json.JsonSerializer
            .Deserialize<List<string>>(raw) ?? new();

        if (!paths.Contains(entry.Path))
        {
            paths.Add(entry.Path);
            CephaInterop.StorageSet(GlobalIndex,
                System.Text.Json.JsonSerializer.Serialize(paths));
        }

        await Task.CompletedTask;
    }

    public Task<IReadOnlyList<CephaFileEntry>> ListAsync(
        string prefix = "", string? mime = null, int limit = 100)
    {
        var raw   = CephaInterop.StorageGet(GlobalIndex) ?? "[]";
        var paths = System.Text.Json.JsonSerializer
            .Deserialize<List<string>>(raw) ?? new();

        var filtered = paths
            .Where(p => string.IsNullOrEmpty(prefix)
                     || p.StartsWith(prefix))
            .Take(limit)
            .Select(p =>
            {
                var fileRaw = CephaInterop.StorageGet(FileKey(p));
                return string.IsNullOrEmpty(fileRaw)
                    ? null
                    : CephaFileEntry.FromKvValue(p, fileRaw);
            })
            .Where(e => e != null
                && (mime == null || e.MimeType == mime))
            .Cast<CephaFileEntry>()
            .ToList();

        return Task.FromResult<IReadOnlyList<CephaFileEntry>>(filtered);
    }

    public Task DeleteAsync(string path)
    {
        CephaInterop.StorageRemove(FileKey(path));
        CephaInterop.StorageRemove(MetaKey(path));
        return Task.CompletedTask;
    }

    public Task<bool> ExistsAsync(string path)
    {
        var raw = CephaInterop.StorageGet(FileKey(path));
        return Task.FromResult(!string.IsNullOrEmpty(raw));
    }

    public async Task<CephaFsStats> GetStatsAsync()
    {
        var files   = await ListAsync(limit: 1000);
        var byMime  = files
            .GroupBy(f => f.MimeType)
            .ToDictionary(g => g.Key, g => g.Count());
        return new CephaFsStats(
            FileCount:   files.Count,
            TotalBytes:  files.Sum(f => f.SizeBytes),
            Environment: "CloudflareKV",
            ByMimeType:  byMime
        );
    }
}
```

### 3.7 `CephaSecretProvider` — الأسرار عبر البيئات

```csharp
public class CephaSecretProvider
{
    private readonly ICephaFileProvider  _fs;
    private readonly Dictionary<string, string> _cache = new();
    private bool _envLoaded = false;

    // ── الأولوية: EnvVar → .env ملف → KV Store ─────────────
    public async Task<string?> GetAsync(string key)
    {
        // 1. Environment Variable (يعمل في كل البيئات)
        var env = System.Environment.GetEnvironmentVariable(key);
        if (!string.IsNullOrEmpty(env)) return env;

        // 2. Cache داخلي
        if (_cache.TryGetValue(key, out var c)) return c;

        // 3. .env ملف (Node.js فقط)
        if (!_envLoaded) await LoadDotEnvAsync();
        if (_cache.TryGetValue(key, out var e)) return e;

        // 4. KV Store مباشرة (Cloudflare)
        var kv = CephaInterop.StorageGet($"cepha:secret:{key}");
        return string.IsNullOrEmpty(kv) ? null : kv;
    }

    private async Task LoadDotEnvAsync()
    {
        _envLoaded = true;
        var entry = await _fs.GetAsync(".env");
        if (entry == null) return;
        foreach (var line in entry.AsText().Split('\n'))
        {
            var t = line.Trim();
            if (t.StartsWith('#') || !t.Contains('=')) continue;
            var eq = t.IndexOf('=');
            _cache[t[..eq].Trim()] =
                t[(eq + 1)..].Trim().Trim('"').Trim('\'');
        }
        CephaInterop.ConsoleLog(
            $"🔐 {_cache.Count} secrets loaded from .env");
    }
}
```

### 4.4 `main.js` — التوجيه حسب الطوبولوجيا

```javascript
// عند discover() — يُحمِّل قرار الطوبولوجيا
const info = await CephaClient.discover();
if (info?.controllersOnServer) {
    window.__cephaMode    = 'server-first';
    window.__mirrorRoutes = new Set(
        (info.mirrorRoutes || []).map(r => r.toLowerCase())
    );
}

// دالة التوجيه المركزية
function resolveTarget(path) {
    if (window.__cephaMode !== 'server-first') return 'browser';
    const base = path.split('?')[0].toLowerCase();
    return window.__mirrorRoutes?.has(base) ? 'browser' : 'server';
}

// SPA Router
document.addEventListener('click', e => {
    const link = e.target.closest('a[href]');
    if (!link) return;
    const href = link.getAttribute('href');
    if (!href || href.startsWith('http') ||
        href.startsWith('#') || /\.\w{2,5}$/.test(href)) return;
    e.preventDefault();
    CephaLoader.startNav();
    history.pushState({}, '', href);

    if (resolveTarget(href) === 'server'
        && window.CephaClient?.connected) {
        // ← مسار الخادم
        delegateToServer('GET', href);
    } else {
        // ← مسار Browser (الوضع الحالي — لا تغيير)
        worker.postMessage({ type: 'navigate', path: href });
    }
});

// التفويض للخادم
function delegateToServer(method, path, data) {
    const opts = { method, headers: { 'Content-Type': 'application/json' } };
    if (data) opts.body = JSON.stringify(data);

    _originalFetch(`${window.CephaClient.serverUrl}${path}`, opts)
        .then(r => r.json())
        .then(res => {
            if (res.statusCode === 302 && res.body) {
                history.pushState({}, '', res.body);
                worker.postMessage({ type: 'navigate', path: res.body });
            } else if (res.body) {
                pushFrame({ selector: '#app',
                            op: 'setInnerHTML', value: res.body });
            }
        })
        .catch(() => {
            // Fallback للـ Worker عند انقطاع الخادم
            worker.postMessage({ type: 'navigate', path });
        });
}
```

## 5. تعزيزات الأمان

*مُصمَّمة خصيصاً لبيئة WASM على الحافة — لا تُعدِّل الهندسة الأساسية.*

### 5.1 جدول التهديدات والتخفيفات المحدَّث

| # | التهديد | الوضع الراهن | التخفيف المقترح |
|---|---------|-------------|----------------|
| **S-01** | XSS عبر `innerHTML` | 🔴 لا sanitization | DOMPurify في `applyFrame()` قبل كل `setInnerHTML` |
| **S-02** | CORS Wildcard `*` | 🔴 `*` ثابت | Origin Allowlist قابل للتهيئة في `CephaRequestPipeline.UseCors()` |
| **S-03** | SSE بدون Auth | 🔴 مفتوح | Token validation قبل `SseConnectionManager.Connect()` |
| **S-04** | OPFS بدون تشفير | 🔴 نص عادي | AES-GCM للملفات الحساسة في `CephaFS` |
| **S-05** | `postMessage` بلا فحص | 🔴 قبول أعمى | Whitelist للأنواع في `worker.onmessage` |
| **S-06** | لا Rate Limiting | 🔴 مفتوح للـ flood | Per-IP counter في `main.mjs` |
| **S-07** | KV Store مشترك | 🟡 بلا isolation | Namespace isolation: `cepha:file:`, `cepha:secret:`, `cepha:session:` |
| **S-08** | `localStorage` snapshot كامل | 🟡 يُرسَل للـ Worker | فلترة `cepha:*` keys فقط |
| **S-09** | غياب CSP | 🔴 لا CSP | Header في `CephaRequestPipeline` + `<meta>` في `index.html` |
| **S-10** | Fingerprint = UserAgent فقط | 🟡 قابل للتزوير | مصادر متعددة: UA + Accept-Language + Canvas hash |
| **S-11** | لا Audit Log | 🔴 لا رصد | `CephaAuditLog` يُسجِّل في `CephaFS` بتشفير |
| **S-12** | لا Authorization في MvcEngine | 🔴 كل المسارات مكشوفة | `[Authorize]` فحص في `MvcEngine.ProcessRequestAsync()` |

```csharp
// CephaRequestPipeline.cs — UseCors() معدَّل
public CephaRequestPipeline UseCors(
    IEnumerable<string>? allowedOrigins = null)
{
    var allowed = new HashSet<string>(
        allowedOrigins ?? new[] { "*" },
        StringComparer.OrdinalIgnoreCase);

    return Use(next => async context =>
    {
        var origin = context.RequestHeaders
            .GetValueOrDefault("origin", "");

        var effectiveOrigin = allowed.Contains("*")
            ? "*"
            : allowed.Contains(origin) ? origin : "";

        if (!string.IsNullOrEmpty(effectiveOrigin))
            context.ResponseHeaders
                ["Access-Control-Allow-Origin"] = effectiveOrigin;

        context.ResponseHeaders["Vary"] = "Origin";

        if (context.Method == "OPTIONS")
        {
            context.StatusCode  = 204;
            context.ResponseBody = "";
            return;
        }

        await next(context);
    });
}
```

#### S-06: Rate Limiting في `main.mjs`

```javascript
// main.mjs — يُضاف قبل معالجة كل طلب
const _rateLimits = new Map();

function checkRateLimit(ip,
    { max = 100, windowMs = 60_000 } = {}) {
    const now = Date.now();
    let   rec = _rateLimits.get(ip)
              ?? { count: 0, resetAt: now + windowMs };

    if (now > rec.resetAt) {
        rec = { count: 0, resetAt: now + windowMs };
    }
    rec.count++;
    _rateLimits.set(ip, rec);
    return { ok: rec.count <= max, remaining: max - rec.count };
}

// داخل createServer handler:
const ip    = req.headers['cf-connecting-ip']
           ?? req.headers['x-forwarded-for']?.split(',')[0]
           ?? req.socket?.remoteAddress
           ?? 'unknown';

const limit = checkRateLimit(ip);
if (!limit.ok) {
    res.writeHead(429, { 'Content-Type': 'application/json',
                         'Retry-After': '60' });
    res.end(JSON.stringify({ error: 'Too Many Requests' }));
    return;
}
```

```csharp
public class CephaAuditLog
{
    private readonly ICephaFileProvider _fs;

    public enum EventType
    {
        Login, Logout, LoginFailed, Lockout,
        FileRead, FileWrite, FileDelete,
        ConfigChange, UnauthorizedAccess
    }

    public async Task RecordAsync(
        EventType type,
        string?   userId    = null,
        string?   path      = null,
        string?   ipAddress = null,
        bool      success   = true,
        string?   details   = null)
    {
        var entry = new
        {
            timestamp = DateTime.UtcNow,
            type      = type.ToString(),
            userId,
            path,
            ip        = ipAddress,
            success,
            details
        };

        var json    = System.Text.Json.JsonSerializer.Serialize(entry);
        var logPath = $"audit/{DateTime.UtcNow:yyyy-MM-dd}.log";

        // إلحاق بملف اليوم
        var existing = await _fs.GetAsync(logPath);
        var content  = existing != null
            ? existing.AsText() + "\n" + json
            : json;

        await _fs.PutAsync(CephaFileEntry.FromText(logPath, content));
    }
}
```

```csharp
// CephaKit.cs — الإصدار المحدَّث الكامل

public class CephaKitOptions
{
    // الموجود — لا تغيير
    public string ConnectionString         { get; set; } = "Data Source=cepha.db";
    public string IdentityConnectionString { get; set; } = "Data Source=cepha_identity.db";
    public string BaseAddress              { get; set; } = "http://localhost/";

    // ← إضافات CEP-001
    public bool                 ControllersOnServer { get; set; } = false;
    public bool                 AutoPersistDb       { get; set; } = true;
    public string               FileBasePath        { get; set; } = ".";
    public CephaFileEnvironment FileEnvironment     { get; set; } =
        DetectEnvironment();
    public IEnumerable<string>? AllowedOrigins      { get; set; } = null;
    public bool                 EnableAuditLog      { get; set; } = false;

    private static CephaFileEnvironment DetectEnvironment() =>
        string.IsNullOrEmpty(
            System.Environment.GetEnvironmentVariable("HOME"))
        ? CephaFileEnvironment.CloudflareKV
        : CephaFileEnvironment.NodeFileSystem;
}

public static IServiceCollection AddCephaKit(
    this IServiceCollection services,
    Action<CephaKitOptions>? configure = null)
{
    var options = new CephaKitOptions();
    configure?.Invoke(options);

    // ── CephaFS ──────────────────────────────────────────────
    ICephaFileProvider fileProvider =
        options.FileEnvironment == CephaFileEnvironment.NodeFileSystem
            ? new NodeFileProvider(options.FileBasePath)
            : new CloudflareKVFileProvider();

    services.AddSingleton<ICephaFileProvider>(fileProvider);
    services.AddSingleton(options);
    services.AddSingleton<CephaSecretProvider>();
    services.AddSingleton<CephaDbFileManager>();

    if (options.EnableAuditLog)
        services.AddSingleton<CephaAuditLog>();

    // ── الموجود — بدون تغيير ────────────────────────────────
    services.AddDbContext<ApplicationDbContext>(db =>
    {
        db.UseSqlite(options.ConnectionString);
        db.UseQueryTrackingBehavior(
            QueryTrackingBehavior.NoTrackingWithIdentityResolution);
    }, ServiceLifetime.Scoped);

    services.AddDbContext<IdentityDbContext>(db =>
    {
        db.UseSqlite(options.IdentityConnectionString);
        db.UseQueryTrackingBehavior(
            QueryTrackingBehavior.NoTrackingWithIdentityResolution);
    }, ServiceLifetime.Scoped);

    services.AddScoped<IWeatherRepository, WeatherRepository>();
    services.AddScoped<ICityRepository,    CityRepository>();
    services.AddSingleton<IPasswordHasher,  PasswordHasher>();
    services.AddScoped<IUserManager,        UserManager>();
    services.AddScoped<IRoleManager,        RoleManager>();
    services.AddScoped<ISignInManager,      SignInManager>();
    services.AddScoped<ISessionStorageService,
                        Services.CephaSessionStorageService>();

    services.AddScoped(sp => new HttpClient
    {
        BaseAddress = new Uri(options.BaseAddress),
        Timeout     = TimeSpan.FromSeconds(30)
    });

    services.AddSingleton<IMvcEngine,   MvcEngine>();
    services.AddSingleton<ISignalREngine>(sp =>
    {
        var engine = new SignalREngine(sp);
        engine.OnClientEvent =
            (hub, method, connId, args) =>
                CephaInterop.DispatchHubEvent(hub, method, connId, args);
        return engine;
    });

    services.AddSingleton<SseConnectionManager>();
    services.AddSingleton<SseMiddleware>();
    services.AddSingleton<CephaServer>();

    return services;
}

public static async Task<IServiceProvider> UseCephaKit(
    this IServiceProvider provider)
{
    CephaInterop.ConsoleLog("🧬 Cepha Kit: initializing...");

    var options   = provider.GetRequiredService<CephaKitOptions>();
    var dbManager = provider.GetRequiredService<CephaDbFileManager>();
    var secrets   = provider.GetRequiredService<CephaSecretProvider>();

    // ← جديد: استعادة DB
    await dbManager.RestoreDbAsync("cepha.db");
    await dbManager.RestoreDbAsync("cepha_identity.db");

    // ← جديد: تحميل الأسرار
    _ = await secrets.GetAsync("JWT_SECRET");

    // الموجود — بدون تغيير
    using (var scope = provider.CreateScope())
    {
        var db = scope.ServiceProvider
            .GetRequiredService<ApplicationDbContext>();
        await db.Database.EnsureCreatedAsync();
    }

    using (var scope = provider.CreateScope())
    {
        var db = scope.ServiceProvider
            .GetRequiredService<IdentityDbContext>();
        await db.Database.EnsureCreatedAsync();
    }

    // ← جديد: persist بعد EnsureCreated على Cloudflare
    if (options.AutoPersistDb &&
        options.FileEnvironment == CephaFileEnvironment.CloudflareKV)
    {
        await dbManager.PersistDbAsync("/tmp/cepha.db");
        await dbManager.PersistDbAsync("/tmp/cepha_identity.db");
    }

    ViewResult.Configure();

    var server = provider.GetRequiredService<CephaServer>();
    server.Start();

    CephaInterop.ConsoleLog(
        $"✅ Cepha Kit ready [{options.FileEnvironment}] " +
        $"[{(options.ControllersOnServer ? "server-first" : "browser-first")}]");

    return provider;
}
```
